// Creating tables
// You can define the tables with full schema names
Table flights {
  id int [pk]
  flight_num varchar
  depart_time datetime
  depart_airport_id int
  arr_time datetime
  arr_airport_id int
  flight_duration time
  co2_emission varchar
  Indexes {
  (id) [pk]
  }
}

Table airports {
  id int [pk]
  iata_code varchar
  name varchar
  city varchar
  Indexes {
    (id) [pk]
  }
}

Table trips {
  id int [pk]
  unique_id varchar
  date_of_scrape date
  price int
  Indexes {
    (id) [pk]
  }
}

Table facilities {
  id int [pk]
  text varchar
  Indexes {
    (id) [pk]
  }
}

Table flights_facilities {
  flight_id int
  facility_id int
}

Table trips_flights{
  trip_id int
  flight_id int
  order_in_trip int
}

Ref: flights.id < trips_flights.flight_id
Ref: trips.id < trips_flights.trip_id
Ref: flights.arr_airport_id > airports.id  
Ref: flights.depart_airport_id > airports.id  
Ref: flights.id < flights_facilities.flight_id 
Ref: facilities.id < flights_facilities.facility_id
